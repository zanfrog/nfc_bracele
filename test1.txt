<script>
  const totalPhotos = 100; // 請把這個數字改成你資料夾裡實際的照片總數！
  const randomImgId = Math.floor(Math.random() * totalPhotos) + 1;
  document.body.style.backgroundImage = `url('background/bg${randomImgId}.png')`;

  const now = new Date();
  const year = now.getFullYear();
  const month = now.getMonth() + 1;
  const date = now.getDate();
  const days = ["星期日", "星期一", "星期二", "星期三", "星期四", "星期五", "星期六"];
  const todayStr = `${month}月${date}日 ${days[now.getDay()]}`;
  const dateKey = `${year}-${month}-${date}`;

  document.getElementById('date-display').innerText = todayStr;

  const btn = document.getElementById('draw-btn');
  const ritualStage = document.getElementById('ritual-stage');
  const resultContainer = document.getElementById('result-container');
  const redrawNote = document.getElementById('redraw-note');

  let alreadyDrawnToday = false;
  try {
    const saved = localStorage.getItem('myLuckyBracelet');
    if (saved) {
      const parsed = JSON.parse(saved);
      if (parsed.savedDate === dateKey) {
        alreadyDrawnToday = true;
        btn.innerText = '輕點蛙蛙，看看今日運勢';
      }
    }
  } catch (e) {}

  function spawnRipple(e) {
    const r = document.createElement('span');
    r.className = 'ripple';
    const rect = btn.getBoundingClientRect();
    const size = Math.max(rect.width, rect.height);
    r.style.width = r.style.height = size + 'px';
    r.style.left = (e.clientX - rect.left - size / 2) + 'px';
    r.style.top = (e.clientY - rect.top - size / 2) + 'px';
    btn.appendChild(r);
    setTimeout(() => r.remove(), 650);
  }

  function spawnPetals() {
    const card = document.querySelector('.card');
    for (let i = 0; i < 6; i++) {
      const p = document.createElement('div');
      p.className = 'petal';
      p.style.left = (10 + Math.random() * 80) + '%';
      p.style.background = i % 2 === 0 ? 'var(--frog)' : 'var(--pond)';
      card.appendChild(p);
      setTimeout(() => {
        p.classList.add('fall');
        p.style.animationDelay = (Math.random() * 0.4) + 's';
      }, 10);
      setTimeout(() => p.remove(), 2800);
    }
  }

  function playReveal(luckyData) {
    document.getElementById('color-box').style.backgroundColor = luckyData.color.hex;
    document.getElementById('color-name').innerHTML = `幸運色：<span class="highlight-color">${luckyData.color.name}</span>`;
    document.getElementById('number-val').innerText = luckyData.num;
    document.getElementById('wish').innerText = "「" + luckyData.sentence.text + "」";

    resultContainer.classList.add('show');
    if (alreadyDrawnToday) redrawNote.style.display = 'block';

    const colorBox = document.getElementById('color-box');
    const colorName = document.getElementById('color-name');
    const seal = document.getElementById('number-seal');
    const wish = document.getElementById('wish');

    colorBox.classList.add('play', 'pop');
    setTimeout(() => { colorName.classList.add('play'); }, 200);
    setTimeout(() => { seal.classList.add('play', 'stamp'); if (navigator.vibrate) navigator.vibrate(60); }, 420);
    setTimeout(() => { wish.classList.add('play'); }, 750);
    setTimeout(() => { spawnPetals(); }, 500);
  }

  btn.addEventListener('click', async function (e) {
    if (navigator.vibrate) navigator.vibrate(200);
    spawnRipple(e);

    btn.style.display = 'none';

    const skipHop = alreadyDrawnToday;

    if (!skipHop) {
      ritualStage.classList.add('active');
    }

    try {
      let luckyData = null;
      const savedData = localStorage.getItem('myLuckyBracelet');

      if (savedData) {
        const parsedData = JSON.parse(savedData);
        if (parsedData.savedDate === dateKey) {
          luckyData = parsedData;
        }
      }

      const fetchPromise = (async () => {
        if (luckyData) return luckyData;
        const colors = await fetch('color.json').then(res => res.json());
        const sentences = await fetch('sentence.json').then(res => res.json());
        const newColor = colors[Math.floor(Math.random() * colors.length)];
        const newSentence = sentences[Math.floor(Math.random() * sentences.length)];
        const newNum = Math.floor(Math.random() * 99) + 1;
        const data = { savedDate: dateKey, color: newColor, sentence: newSentence, num: newNum };
        localStorage.setItem('myLuckyBracelet', JSON.stringify(data));
        return data;
      })();

      const minWait = skipHop ? 250 : 1050;
      const [finalData] = await Promise.all([
        fetchPromise,
        new Promise(res => setTimeout(res, minWait))
      ]);

      ritualStage.classList.remove('active');
      playReveal(finalData);

    } catch (error) {
      console.error("讀取失敗:", error);
      ritualStage.classList.remove('active');
      resultContainer.classList.add('show');
      document.getElementById('wish').innerText = "讀取資料時發生錯誤，請檢查檔案路徑。";
      document.getElementById('wish').classList.add('play');
    }
  });
</script>
