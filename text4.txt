  function loadJSON(key, fallback){
    try{ const v = localStorage.getItem(key); return v ? JSON.parse(v) : fallback; }
    catch(e){ return fallback; }
  }
  function saveJSON(key, val){
    try{ localStorage.setItem(key, JSON.stringify(val)); }catch(e){}
  }

  let profile = loadJSON('lb_profile', null);
  let history = loadJSON('lb_history', []);

  // ---------- DOM ----------
  const btn = document.getElementById('draw-btn');
  const ritualStage = document.getElementById('ritual-stage');

  const modeLabel = document.getElementById('mode-label');
  const specTag = document.getElementById('spec-tag');

  const ssrBanner = document.getElementById('ssr-banner');

      modeLabel.innerHTML = `目前狀態：<b>${modeNames[profile.mode] || '純粹求平安'}</b>`;
    } else {
      modeLabel.innerHTML = '';
    }
  }
  refreshModeLabel();

  // ---------- 已抽過今日判斷 ----------
  let todayData = loadJSON('lb_today', null);
  let alreadyDrawnToday = !!(todayData && todayData.savedDate === dateKey);
  if(alreadyDrawnToday) btn.innerText = '輕點蛙蛙，看看今日運勢';

  // ---------- 首次使用：選擇狀態 ----------
  if(!profile){
    onboardOverlay.classList.add('show');
  }
  document.querySelectorAll('.mode-card').forEach(card=>{
    card.addEventListener('click', ()=>{
      profile = { mode: card.dataset.mode };
      saveJSON('lb_profile', profile);
      refreshModeLabel();
      onboardOverlay.classList.remove('show');
    });
  });

  function spawnRipple(e){
    const r = document.createElement('span');
    r.className = 'ripple';
    const rect = btn.getBoundingClientRect();
    const size = Math.max(rect.width, rect.height);
    r.style.width = r.style.height = size + 'px';
    r.style.left = (e.clientX - rect.left - size/2) + 'px';
    r.style.top = (e.clientY - rect.top - size/2) + 'px';
    btn.appendChild(r);
    setTimeout(()=>r.remove(), 650);
  }

  function startMeteorShower(){
    meteorField.innerHTML = '';
    meteorField.classList.add('active');
    document.body.classList.add('ssr-mode');

  // ---------- 資料抓取（分類金句 + 特殊庫） ----------
  async function fetchPool(mode){
    const colors = await fetch('color.json').then(r=>r.json());
    let sentenceData;
    try{ sentenceData = await fetch('sentence.json').then(r=>r.json()); }
    catch(e){ sentenceData = []; }

    let pool;
    if(Array.isArray(sentenceData)){
      pool = sentenceData; // 舊格式：單一陣列，所有狀態共用
    } else {
      pool = sentenceData[mode] || sentenceData.all || Object.values(sentenceData).flat();
    }
    return { colors, pool };
  }

  async function fetchSSRPool(){
    try{
      const data = await fetch('ssr.json').then(r=>r.json());
      return Array.isArray(data) ? data : (data.ssr || []);
    }catch(e){
      return [
        { text: "SSR 級超幸運：今天宜買樂透，宜大膽告白，宜做任何你猶豫已久的事" },
        { text: "流星劃過的瞬間，好運已經悄悄落在你的手串上了" },
        { text: "百分之一的機率被你抽中，代表今天注定不平凡" }
      ];
    }
  }

  // ---------- 抽籤主邏輯 ----------
  btn.addEventListener('click', async function(e){
    if(navigator.vibrate) navigator.vibrate(200);
    spawnRipple(e);
    btn.style.display = 'none';

    const skipHop = alreadyDrawnToday;
    if(!skipHop) ritualStage.classList.add('active');

    try{
      let luckyData = null;
      if(alreadyDrawnToday && todayData){
        luckyData = todayData;
      }

      const workPromise = (async ()=>{
        if(luckyData) return luckyData;

        const mode = (profile && profile.mode) || 'peace';
        const isSSR = Math.random() < 0.01;
        const { colors, pool } = await fetchPool(mode);

        const newColor = colors[Math.floor(Math.random()*colors.length)];
        const newNum = Math.floor(Math.random()*99) + 1;

        let newSentence;
        if(isSSR){
          const ssrPool = await fetchSSRPool();
          newSentence = ssrPool[Math.floor(Math.random()*ssrPool.length)];
        } else {
          newSentence = pool[Math.floor(Math.random()*pool.length)];
        }

        const data = { savedDate: dateKey, color: newColor, sentence: newSentence, num: newNum, mode, isSSR };
        saveJSON('lb_today', data);

        history.push({ date: dateKey, colorName: newColor.name, colorHex: newColor.hex, num: newNum, isSSR });
        if(history.length > 120) history = history.slice(-120);
        saveJSON('lb_history', history);

        return data;
      })();

      const minWait = skipHop ? 200 : 900;
      const [finalData] = await Promise.all([workPromise, new Promise(res=>setTimeout(res, minWait))]);

      ritualStage.classList.remove('active');
      todayData = finalData;
      playReveal(finalData);

    }catch(err){
      console.error('讀取失敗:', err);
      ritualStage.classList.remove('active');
      resultContainer.classList.add('show');
      document.getElementById('wish').innerText = '讀取資料時發生錯誤，請檢查檔案路徑。';
      document.getElementById('wish').classList.add('play');
    }
  });

  function renderDashboard(){
    const monthPrefix = `${year}-${String(month).padStart(2,'0')}`;
    const monthCount = history.filter(h => h.date.startsWith(monthPrefix)).length;
    document.getElementById('stat-month-count').innerText = monthCount + ' 天';
    document.getElementById('stat-total-count').innerText = history.length + ' 天';

    const ssrCount = history.filter(h => h.isSSR).length;
    document.getElementById('stat-ssr-count').innerText = ssrCount + ' 次';

    const freq = {};
    history.forEach(h=>{
      if(!freq[h.colorName]) freq[h.colorName] = { count:0, hex:h.colorHex };
      freq[h.colorName].count++;
    });
    let topColor = null, topCount = 0;
    Object.keys(freq).forEach(name=>{
      if(freq[name].count > topCount){ topCount = freq[name].count; topColor = name; }
    });
    const topColorEl = document.getElementById('stat-top-color');
    if(topColor){
      topColorEl.innerHTML = `<span class="stat-dot" style="background:${freq[topColor].hex}"></span>${topColor}`;
    } else {
      topColorEl.innerHTML = `<span class="stat-dot" style="background:#ccc;"></span>尚無紀錄`;
    }

    const trendRow = document.getElementById('trend-row');
    trendRow.innerHTML = '';
    const recent = history.slice(-14);
    if(recent.length === 0){
      trendRow.innerHTML = '<div style="font-size:0.72rem;color:var(--mist);">還沒有抽籤紀錄，抽完今日能量後回來看看吧</div>';
    } else {
      recent.forEach(h=>{
        const wrap = document.createElement('div');
        wrap.className = 'trend-bar';
        const bar = document.createElement('div');
        bar.className = 'bar' + (h.isSSR ? ' ssr' : '');
        const pct = Math.max(6, Math.round((h.num/99)*100));
        bar.style.height = pct + '%';
        const numLabel = document.createElement('div');
        numLabel.className = 'num';
        numLabel.innerText = h.num;
        wrap.appendChild(bar);
        wrap.appendChild(numLabel);
        trendRow.appendChild(wrap);
      });
    }
  }
